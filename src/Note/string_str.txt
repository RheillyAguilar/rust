let mut s: String = String::from("");
s.push_str("hello, world");
s.push('!');
assert_eq!(s, "hello, world!");

s.push(',');
s.push_str(" world");

println!("{}", s);

let mut s: String = String::from("I like dogs");
let s1 = s.replace("dogs", "cats");
assert_eq!(s1, "I like cats");
